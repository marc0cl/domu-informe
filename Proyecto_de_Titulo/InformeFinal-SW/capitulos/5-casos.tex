% ====== 4.1 Historias de usuario ======
\section{Historias de usuario}
\begin{itemize}
  \item Como \textbf{residente}, quiero recibir notificaciones de mis encomiendas para retirarlas a tiempo.
  \item Como \textbf{administrador}, quiero generar reportes financieros para tomar decisiones de gestión.
  \item Como \textbf{conserje}, quiero registrar ingreso/salida de visitas para resguardar la seguridad.
\end{itemize}
